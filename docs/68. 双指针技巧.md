```
从已排序数组中删除指定值
```

```cpp
class Solution {
public:
  int removeElement(std::vector<int>& nums, int val) {
    int slow = 0, fast = 0;
    while (fast < nums.size()) {
      if (nums[fast] != val) {
        nums[slow++] = nums[fast];
      }
      fast++;
    }

    return slow;
  }
};
```
### 移动`0`, 将`0`全部移动到数组末尾
```cpp
class Solution {
public:
  void moveZeroes(std::vector<int>& nums) {
    int slow = 0, fast = 0;
    while (fast < nums.size()) {
      if (nums[fast] != 0) {
        swap(nums[slow++], nums[fast]);
      }
      fast++;
    }
  }
};
```
